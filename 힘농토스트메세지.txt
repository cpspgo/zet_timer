function rand(arr){
  return arr[Math.floor(Math.random()*arr.length)];
}

function ppHide(){
  document.getElementById('purchase-popup').classList.remove('show');
}

window.ppHide = ppHide;

function ppShow(){

  document.getElementById('pp-location').textContent = rand(locations);
  document.getElementById('pp-icon').textContent = rand(emojis);

  var el = document.getElementById('purchase-popup');

  el.classList.remove('show');

  setTimeout(function(){
    el.classList.add('show');
  },80);

  setTimeout(function(){
    ppHide();
  },3000);

}

function ppGetIntervalRange(){

  var h = new Date().getHours();

  if (h >= 0 && h < 3) return [46,74];
  if (h >= 3 && h < 6) return [83,134];
  if (h >= 6 && h < 9) return [37,60];
  if (h >= 9 && h < 12) return [17,27];
  if (h >= 12 && h < 15) return [13,21];
  if (h >= 15 && h < 18) return [8,13];
  if (h >= 18 && h < 21) return [14,23];

  return [27,44];

}

function schedule(){

  var r = ppGetIntervalRange();

  var delaySec = Math.random()*(r[1]-r[0])+r[0];

  setTimeout(function(){
    ppShow();
    schedule();
  },delaySec*1000);

}
